Method 1: Picture Element (Recommended)

    <source srcset="images/optimized/logo.webp" type="image/webp">
    <img src="images/optimized/logo.jpg" alt="ST Sportz Logo">
</picture>

Method 2: CSS Background with Fallback

.hero-image {
    background-image: url('images/optimized/hero.jpg');
}

@supports (background-image: url('images/optimized/hero.webp')) {
    .hero-image {
        background-image: url('images/optimized/hero.webp');
    }
}

Method 3: JavaScript Detection

<script>
function supportsWebP() {
    const elem = document.createElement('canvas');
    return elem.getContext && elem.getContext('2d') 
        ? elem.toDataURL('image/webp').indexOf('data:image/webp') === 0 
        : false;
}

if (supportsWebP()) {
    // Replace all images with WebP versions
    const images = document.querySelectorAll('img[data-webp]');
    images.forEach(img => {
        img.src = img.dataset.webp;
    });
}
</script>
